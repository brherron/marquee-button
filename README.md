### React Marquee Button

## install

```
npm install marquee-button
```

## usage

```js
import { Button } from 'marquee-button'

function Home() {

  return (
    <Button path="https://beauherrondev.com" text="Super Cool Website" color="pink"/>
  )
}

export default Home;
```
